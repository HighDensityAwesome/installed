	## 1 ##
		
		sudo su		

		sudo dnf install kernel-devel kernel-headers gcc dkms acpid

		echo "blacklist nouveau" >> /etc/modprobe.d/blacklist.conf
	
		nano /etc/sysconfig/grub
		Edit /etc/sysconfig/grub:
			append ‘rd.driver.blacklist=nouveau’ to end of ‘GRUB_CMDLINE_LINUX=”…”‘

		#UPDATE GRUB2 CONF
			## BIOS ##
		grub2-mkconfig -o /boot/grub2/grub.cfg
			## UEFI ## LIKELY USING THIS ##
		grub2-mkconfig -o /boot/efi/EFI/fedora/grub.cfg
	
		#REMOVE XORG-X11-DRV-NOUVEAU
		dnf remove xorg-x11-drv-nouveau

		#GEN INITRAMFS
			## Backup old initramfs nouveau image ##
		mv /boot/initramfs-$(uname -r).img /boot/initramfs-$(uname -r)-nouveau.img
			## Create new initramfs image ##
		dracut /boot/initramfs-$(uname -r).img $(uname -r)

		#REBOOT TO RUNLEVEL 3(with grub edit or following):
		systemctl set-default multi-user.target

		reboot
